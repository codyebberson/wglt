+game.entities.push(player);

+const messageLog = new wglt.MessageLog(game.gui, new wglt.Rect(1, 1, 100, 100));
+messageLog.add('Hello world!');
+messageLog.add('Use arrow keys to move');
+game.gui.add(messageLog);
+
